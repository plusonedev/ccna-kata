Setup DHCP pool for 0/1 - Solution


> enable
# config term
(config)# ip dhcp pool primary-pool
(dhcp-config)# network 10.0.0.0 /24
(dhcp-config)# domain-name testnet.local
(dhcp-config)# dns-server 8.8.8.8 8.8.4.4
(dhcp-config)# default-router 10.0.0.1
(dhcp-config)# lease 7
(dhcp-config)# exit
(config)# ip dhcp excluded-add 10.0.0.1 10.0.0.99
